---
title: "List Invoices"
api: "GET https://api.getredo.com/v2.2/stores/{storeId}/invoices"
description: "List all invoices for a store"
---

## Path Parameters

<ParamField path="storeId" type="string" required>
  The unique identifier for your store
</ParamField>

## Query Parameters

<ParamField query="page-size" type="integer">
  Number of invoices to return per page
</ParamField>

<ParamField query="page-continue" type="string">
  Cursor token from the previous response's `X-Page-Next` header to fetch the next page
</ParamField>

<ParamField query="updated-at-min" type="string">
  Filter invoices updated after this timestamp (ISO 8601 format)
</ParamField>

<ParamField query="updated-at-max" type="string">
  Filter invoices updated before this timestamp (ISO 8601 format)
</ParamField>

## Response

<ResponseField name="invoices" type="array">
  Array of invoice objects

  <Expandable title="Invoice object">
    <ResponseField name="id" type="string">
      Invoice identifier
    </ResponseField>

    <ResponseField name="storeId" type="string">
      Store identifier
    </ResponseField>

    <ResponseField name="periodStart" type="string">
      ISO 8601 timestamp of invoice period start
    </ResponseField>

    <ResponseField name="periodEnd" type="string">
      ISO 8601 timestamp of invoice period end
    </ResponseField>

    <ResponseField name="status" type="string">
      Invoice status (e.g., `pending`, `paid`, `overdue`)
    </ResponseField>

    <ResponseField name="amount" type="object">
      Invoice amount

      <Expandable title="Money object">
        <ResponseField name="amount" type="string">
          Amount as a string (e.g., "150.00")
        </ResponseField>

        <ResponseField name="currency" type="string">
          Currency code (e.g., "USD")
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="dueDate" type="string">
      ISO 8601 timestamp of payment due date
    </ResponseField>

    <ResponseField name="createdAt" type="string">
      ISO 8601 timestamp of when the invoice was created
    </ResponseField>

    <ResponseField name="updatedAt" type="string">
      ISO 8601 timestamp of when the invoice was last updated
    </ResponseField>
  </Expandable>
</ResponseField>

## Response Headers

<ResponseField name="X-Page-Next" type="string">
  Cursor token to fetch the next page of results
</ResponseField>
